def is_palindrome(input_str):

\begin{frame}[fragile]
    \frametitle{Exercise}
    Based on Euclid's algorithm:
        $gcd(a,b)=gcd(b,b\%a)$\\
    gcd function can be written as:
    \begin{lstlisting}
    def gcd(a, b):
      if a%b == 0: return b
      return gcd(b, a%b)
    \end{lstlisting}
