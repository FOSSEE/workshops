\begin{frame}[fragile]
    \frametitle{Exercise: Lorenz equation}
    \begin{columns}
        \column{0.25\textwidth}
        \begin{eqnarray*}
        \frac{d x}{dt} &=& s (y-x)\\
        \frac{d y}{d t} &=& rx -y -xz\\
        \frac{d z}{d t} &=& xy - bz\\
        \end{eqnarray*}
        \column{0.25\textwidth}
        Let $s=10,$
        $r=28,$ 
        $b=8./3.$
    \end{columns}
    \structure{\Large Region of interest}
  \begin{lstlisting}
x, y, z = mgrid[-50:50:20j,-50:50:20j,
                -10:60:20j]
  \end{lstlisting}
\inctime{20}

\end{frame}
\begin{frame}[fragile]
    \frametitle{Solution}
  \begin{lstlisting}
def lorenz(x,y,z,s=10.,r=28.,b=8./3.):
    u = s*(y-x)
    v = r*x-y-x*z
    w = x*y-b*z
    return u,v,w
x,y,z = mgrid [-50:50:20j,-50:50:20j,
                    -10:60:20j ]
u,v,w = lorenz( x , y , z )
# Your plot here
#
mlab.show()

  \end{lstlisting}
\end{frame}
