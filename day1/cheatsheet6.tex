\section{Solving linear equations}
\begin{verbatim}
    In []: A = array([[3,2,-1],
                      [2,-2,4],                   
                      [-1, 0.5, -1]])
    In []: b = array([[1], [-2], [0]])
    In []: x = solve(A, b)
    In []: Ax = dot(A,x)
    In []: allclose(Ax, b)
    Out[]: True
\end{verbatim}
