\begin{frame}[fragile]

\begin{frame}[fragile]

\begin{lstlisting}
In []: TSq = []

In []: for t in T:
 ....:     TSq.append(t*t)

In []: plot(L, TSq)
Out[]: [<matplotlib.lines.Line2D object at 0xa5b05ac>]
\end{lstlisting}
\end{frame}
